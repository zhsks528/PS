def solution(n, works):
    if n > sum(works):
        return 0
    
    heap = []
    
    for i in works:
        heappush(heap, (-i, i))
    
    for i in range(n):
        update = heappop(heap)
        value = update[1] - 1
        heappush(heap, (-value, value))
    
    works = [num[1]**2 for num in heap]

    return sum(works)
