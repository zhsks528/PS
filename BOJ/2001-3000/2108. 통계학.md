import sys
from collections import Counter

# 1
def _sum(nums):
    result = round(sum(nums) / len(nums))

    return result

# 2
def _mid(nums):
    nums.sort()
    mid_idx = len(nums) // 2

    result = nums[mid_idx]

    return result

# 3
def _mode(nums):

    c = Counter(nums)
    order = c.most_common()
    _max = order[0][1]
    modes = []

    for num in order:
        if num[1] == _max:
            modes.append(num[0])

    if len(modes) == 1:
        return modes[0]
    else:
        return modes[1]

# 4
def _range(nums):
    nums_max = max(nums)
    nums_min = min(nums)

    return nums_max - nums_min


input = sys.stdin.readline
n = int(input())
nums = [int(input()) for _ in range(n)]

print(_sum(nums))
print(_mid(nums))
print(_mode(nums))
print(_range(nums))
