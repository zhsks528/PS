from heapq import heappush, heappop

n = int(input())
arr = [list(map(int, input().split())) for _ in range(n)]
heap = []

for i in arr:
    for j in i:
        heappush(heap, j)

sorted_nums = []

while heap:
    sorted_nums.append(heappop(heap))

res = sorted(sorted_nums, reverse=True)
