# 1
for i in range(n):
    if i == del_node or parents[i] == del_node:
        continue

    if parents[i] in tree:
        tree[parents[i]].append(i)
    else:
        tree[parents[i]] = [i]

# 2

    q = [-1]
else:
    q = []

# 3

    node = q.pop()

    if node not in tree:
        answer += 1
    else:
        q += tree[node]
