```python
import sys

n, m = map(int, sys.stdin.readline().split())
parent = [i for i in range(n+1)]

def find(x):
    if parent[x] != x:
        parent[x] = find(parent[x])

    return parent[x]

def union(a, b):
    root1 = find(a)
    root2 = find(b)

    if root1 != root2:
        parent[root2] = root1

for i in range(m):
    cmd, a, b = map(int, sys.stdin.readline().split())

    if cmd == 0:
        union(a, b)
    else:
        if find(a) == find(b):
            print('YES')
        else:
            print('NO')
```
