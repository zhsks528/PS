from collections import deque

n, k = map(int, input().split())
visited = [False] * 100001

q = deque()

def bfs():
    q.append((n, 0))

    while q:
        x, c = q.popleft()

        if not(visited[x]):
            visited[x] = True

            if x == k:
                print(c)
                break

            if x - 1 >= 0:
                q.append((x - 1, c+1))
            if x + 1 <= 100000:
                q.append((x + 1, c+1))
            if x * 2 <= 100000:
                q.append((x*2, c+1))

bfs()
