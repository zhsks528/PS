# 1. push
# 2. pop
# 3. size
# 4. empty
# 5. front
# 6. back
import sys

# 1
def push(x):
    q.append(x)

# 2
def pop():
    if len(q) == 0:
        print(-1)
    else:
        first = q.pop(0)
        print(first)

# 3
def size():
    print(len(q))

# 4
def empty():
    if len(q) > 0:
        print(0)
    else:
        print(1)

# 5
def front():
    if len(q) > 0:
        print(q[0])
    else:
        print(-1)

# 6
def back():
    if len(q) > 0:
        print(q[-1])
    else:
        print(-1)

input = sys.stdin.readline
n = int(input())
q = []

for _ in range(n):
    data = list(input().split())
    cmd = data[0]

    if cmd == 'push':
        push(data[1])
    elif cmd == 'pop':
        pop()
    elif cmd == 'size':
        size()
    elif cmd == 'empty':
        empty()
    elif cmd == 'front':
        front()
    elif cmd == 'back':
        back()
