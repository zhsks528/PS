from collections import deque

n, m = map(int, input().split())
a = [list(input().strip()) for _ in range(n)]
ck = [[[[False] * m for _ in range(n)] for _ in range(m)] for _ in range(n)]
q = deque()

dx = [1, 0, -1, 0]
dy = [0, 1, 0, -1]


def init():
    _rx, _ry, _bx, _by = [0] * 4

    for i in range(n):
        for j in range(m):
            if a[i][j] == 'R':
                _rx, _ry = i, j
            elif a[i][j] == 'B':
                _bx, _by = i, j

    q.append((_rx, _ry, _bx, _by, 0))
    ck[_rx][_ry][_bx][_by] = True


def move(x, y, dx, dy, cnt):

    while a[x+dx][y+dy] != '#' and a[x][y] != 'O':
        x += dx
        y += dy
        cnt += 1

    return x, y, cnt


def bfs():
    while q:
        rx, ry, bx, by, d = q.popleft()

        if d >= 10:
            break

        for i in range(4):
            nrx, nry, nrc = move(rx, ry, dx[i], dy[i], 0)
            nbx, nby, nbc = move(bx, by, dx[i], dy[i], 0)

            if a[nbx][nby] == 'O':
                continue

            if a[nrx][nry] == 'O':
                print(1)
                return

            if nrx == nbx and nry == nby:
                if nrc > nbc:
                    nrx -= dx[i]
                    nry -= dy[i]
                else:
                    nbx -= dx[i]
                    nby -= dy[i]

            if not ck[nrx][nry][nbx][nby]:
                ck[nrx][nry][nbx][nby] = True
                q.append((nrx, nry, nbx, nby, d+1))

    print(0)

init()
bfs()
